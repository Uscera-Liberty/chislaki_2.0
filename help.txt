using Npgsql;
using System;
using System.Data;
using System.Drawing;
using System.Windows.Forms;
using Excel = Microsoft.Office.Interop.Excel;

namespace MyZadacha
{
    public class ReportExcel : Form
    {
        private NpgsqlConnection con;
        private DateTimePicker dtpStart;
        private DateTimePicker dtpEnd;
        private Button btnExport;
        private Button btnCancel;
        private Label lblInfo;

        public ReportExcel(NpgsqlConnection connection)
        {
            con = connection;
            InitializeComponent();
        }

        private void InitializeComponent()
        {
            this.Text = "Налоговый отчёт за период (Excel)";
            this.Size = new Size(450, 200);
            this.StartPosition = FormStartPosition.CenterParent;
            this.FormBorderStyle = FormBorderStyle.FixedDialog;
            this.MaximizeBox = false;

            lblInfo = new Label
            {
                Text = "Выберите период для формирования отчёта:",
                Location = new Point(20, 20),
                Size = new Size(400, 25),
                Font = new Font("Arial", 10, FontStyle.Bold)
            };

            Label lblStart = new Label { Text = "Начало периода:", Location = new Point(30, 60), Size = new Size(100, 25) };
            dtpStart = new DateTimePicker { Location = new Point(140, 60), Size = new Size(130, 25), Format = DateTimePickerFormat.Short };

            Label lblEnd = new Label { Text = "Конец периода:", Location = new Point(30, 95), Size = new Size(100, 25) };
            dtpEnd = new DateTimePicker { Location = new Point(140, 95), Size = new Size(130, 25), Format = DateTimePickerFormat.Short };

            btnExport = new Button
            {
                Text = "Сформировать и выгрузить",
                Location = new Point(290, 75),
                Size = new Size(130, 35),
                BackColor = Color.LightGreen
            };
            btnExport.Click += BtnExport_Click;

            btnCancel = new Button
            {
                Text = "Отмена",
                Location = new Point(290, 120),
                Size = new Size(130, 30)
            };
            btnCancel.Click += (s, e) => Close();

            Controls.Add(lblInfo);
            Controls.Add(lblStart);
            Controls.Add(dtpStart);
            Controls.Add(lblEnd);
            Controls.Add(dtpEnd);
            Controls.Add(btnExport);
            Controls.Add(btnCancel);
        }

        private void BtnExport_Click(object sender, EventArgs e)
        {
            ExportToExcel();
        }

        private void ExportToExcel()
        {
            DataTable reportData = GetReportData();
            if (reportData == null || reportData.Rows.Count == 0)
            {
                MessageBox.Show("Нет данных за выбранный период.", "Внимание",
                    MessageBoxButtons.OK, MessageBoxIcon.Warning);
                return;
            }

            Excel.Application excelApp = null;
            Excel.Workbook workbook = null;
            Excel.Worksheet worksheet = null;

            try
            {
                excelApp = new Excel.Application();
                excelApp.DisplayAlerts = false;
                workbook = excelApp.Workbooks.Add();
                worksheet = (Excel.Worksheet)workbook.Worksheets[1];
                worksheet.Name = "Налоговый отчёт";

                // Заголовок
                worksheet.Cells[1, 1] = "НАЛОГОВЫЙ ОТЧЁТ ПО ПОДОТЧЁТНЫМ СУММАМ";
                Excel.Range titleRange = worksheet.Range[worksheet.Cells[1, 1], worksheet.Cells[1, 3]];
                titleRange.Merge();
                titleRange.Font.Bold = true;
                titleRange.Font.Size = 14;
                titleRange.HorizontalAlignment = Excel.XlHAlign.xlHAlignCenter;

                // Период и дата формирования
                string startStr = dtpStart.Value.ToString("dd.MM.yyyy");
                string endStr = dtpEnd.Value.ToString("dd.MM.yyyy");
                worksheet.Cells[3, 1] = $"Период: {startStr} – {endStr}";
                worksheet.Cells[3, 1].Font.Bold = true;
                worksheet.Cells[4, 1] = $"Дата формирования: {DateTime.Now:dd.MM.yyyy HH:mm}";

                // Шапка таблицы
                int hRow = 6;
                string[] cols = { "Сотрудник", "Сумма расходов (руб.)", "Выдано из кассы (руб.)", "Облагаемая налогом сумма (руб.)", "Сумма налога 13% (руб.)" };
                for (int i = 0; i < cols.Length; i++)
                {
                    worksheet.Cells[hRow, i + 1] = cols[i];
                    Excel.Range cell = worksheet.Cells[hRow, i + 1];
                    cell.Font.Bold = true;
                    cell.Interior.Color = Color.FromArgb(68, 114, 196);
                    cell.Font.Color = Color.White;
                    cell.HorizontalAlignment = Excel.XlHAlign.xlHAlignCenter;
                }

                // Заполнение данных
                int row = hRow + 1;
                decimal totalSpent = 0;
                decimal totalPaid = 0;
                decimal totalTaxable = 0;
                decimal totalTax = 0;

                foreach (DataRow dr in reportData.Rows)
                {
                    string employeeName = dr["employee_name"].ToString();
                    decimal spent = Convert.ToDecimal(dr["total_spent"]);
                    decimal paid = Convert.ToDecimal(dr["total_paid"]);
                    decimal taxable = Convert.ToDecimal(dr["taxable_sum"]);
                    decimal tax = Convert.ToDecimal(dr["tax_sum"]);

                    worksheet.Cells[row, 1] = employeeName;
                    worksheet.Cells[row, 2] = spent;
                    worksheet.Cells[row, 3] = paid;
                    worksheet.Cells[row, 4] = taxable;
                    worksheet.Cells[row, 5] = tax;

                    totalSpent += spent;
                    totalPaid += paid;
                    totalTaxable += taxable;
                    totalTax += tax;

                    if (row % 2 == 0)
                    {
                        Excel.Range range = worksheet.Range[worksheet.Cells[row, 1], worksheet.Cells[row, 5]];
                        range.Interior.Color = Color.FromArgb(235, 241, 251);
                    }
                    row++;
                }

                // Итоговая строка
                int tRow = row + 1;
                worksheet.Cells[tRow, 1] = "ИТОГО:";
                worksheet.Cells[tRow, 1].Font.Bold = true;
                worksheet.Cells[tRow, 2] = totalSpent;
                worksheet.Cells[tRow, 2].Font.Bold = true;
                worksheet.Cells[tRow, 3] = totalPaid;
                worksheet.Cells[tRow, 3].Font.Bold = true;
                worksheet.Cells[tRow, 4] = totalTaxable;
                worksheet.Cells[tRow, 4].Font.Bold = true;
                worksheet.Cells[tRow, 5] = totalTax;
                worksheet.Cells[tRow, 5].Font.Bold = true;

                Excel.Range totalRange = worksheet.Range[worksheet.Cells[tRow, 1], worksheet.Cells[tRow, 5]];
                totalRange.Interior.Color = Color.FromArgb(255, 230, 153);

                // Формат чисел
                Excel.Range numberRange = worksheet.Range[worksheet.Cells[hRow + 1, 2], worksheet.Cells[tRow, 5]];
                numberRange.NumberFormat = "#,##0.00";

                // Автоширина
                worksheet.Columns[1].ColumnWidth = 30;
                worksheet.Columns[2].ColumnWidth = 18;
                worksheet.Columns[3].ColumnWidth = 18;
                worksheet.Columns[4].ColumnWidth = 22;
                worksheet.Columns[5].ColumnWidth = 18;

                // Сохранение
                string path = System.IO.Path.Combine(
                    Environment.GetFolderPath(Environment.SpecialFolder.Desktop),
                    $"TaxReport_{DateTime.Now:yyyyMMdd_HHmm}.xlsx");

                workbook.SaveAs(path);
                MessageBox.Show($"Файл сохранён:\n{path}", "Готово",
                    MessageBoxButtons.OK, MessageBoxIcon.Information);
                System.Diagnostics.Process.Start(new System.Diagnostics.ProcessStartInfo
                { FileName = path, UseShellExecute = true });
                Close();
            }
            catch (Exception ex)
            {
                MessageBox.Show("Ошибка Excel: " + ex.Message, "Ошибка",
                    MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
            finally
            {
                if (workbook != null) workbook.Close(false);
                if (excelApp != null) excelApp.Quit();
                if (worksheet != null) System.Runtime.InteropServices.Marshal.ReleaseComObject(worksheet);
                if (workbook != null) System.Runtime.InteropServices.Marshal.ReleaseComObject(workbook);
                if (excelApp != null) System.Runtime.InteropServices.Marshal.ReleaseComObject(excelApp);
            }
        }

        private DataTable GetReportData()
        {
            DataTable result = new DataTable();
            try
            {
                DateTime startDate = dtpStart.Value.Date;
                DateTime endDate = dtpEnd.Value.Date;

                // Запрос без cashadvance
                // Налог считается как 13% от разницы между total_spent и total_paid
                // (если total_spent > total_paid, то облагается разница)
                string sql = @"
                    SELECT 
                        e.name AS employee_name,
                        SUM(ar.total_spent) AS total_spent,
                        SUM(ar.total_paid) AS total_paid,
                        SUM(
                            CASE 
                                WHEN ar.total_spent > ar.total_paid 
                                THEN ar.total_spent - ar.total_paid 
                                ELSE 0 
                            END
                        ) AS taxable_sum,
                        SUM(
                            CASE 
                                WHEN ar.total_spent > ar.total_paid 
                                THEN (ar.total_spent - ar.total_paid) * 0.13 
                                ELSE 0 
                            END
                        ) AS tax_sum
                    FROM advancereport ar
                    JOIN employee e ON e.id = ar.employee_id
                    WHERE ar.report_date::date BETWEEN :startDate AND :endDate
                    GROUP BY e.name
                    ORDER BY e.name";

                NpgsqlCommand cmd = new NpgsqlCommand(sql, con);
                cmd.Parameters.AddWithValue("startDate", startDate);
                cmd.Parameters.AddWithValue("endDate", endDate);
                NpgsqlDataAdapter da = new NpgsqlDataAdapter(cmd);
                da.Fill(result);
            }
            catch (Exception ex)
            {
                MessageBox.Show("Ошибка загрузки данных: " + ex.Message, "Ошибка",
                    MessageBoxButtons.OK, MessageBoxIcon.Error);
                return null;
            }
            return result;
        }
    }
}
