        private void BtnGenerate_Click(object sender, EventArgs e)

        private void LoadReport()
        {
            try
            {
                // Запрос: для каждого сотрудника за период суммируем суммы отчётов, где дата отчёта раньше даты выдачи
                // Упрощённо: считаем налогооблагаемую базу как сумму total_spent по отчётам, у которых report_date < (дата выдачи)
                // Но правильнее: для каждого отчёта проверить наличие выдачи позже. Здесь делаем агрегат по сотрудникам.
                string sql = @"
                    WITH report_tax AS (
                        SELECT 
                            ar.employee_id,
                            e.name AS employee_name,
                            ar.total_spent,
                            EXISTS (
                                SELECT 1 FROM cashadvance ca 
                                WHERE ca.employee_id = ar.employee_id 
                                AND ca.date_issued > ar.report_date
                            ) AS has_advance_after_report
                        FROM advancereport ar
                        JOIN employee e ON e.id = ar.employee_id
                        WHERE ar.report_date::date BETWEEN :startDate AND :endDate
                    )
                    SELECT 
                        employee_name,
                        SUM(CASE WHEN has_advance_after_report THEN total_spent ELSE 0 END) AS taxable_sum,
                        SUM(CASE WHEN has_advance_after_report THEN total_spent * 0.13 ELSE 0 END) AS tax_sum
                    FROM report_tax
                    GROUP BY employee_name
                    ORDER BY employee_name";

                NpgsqlCommand cmd = new NpgsqlCommand(sql, con);
                cmd.Parameters.AddWithValue("startDate", dtpStart.Value.Date);
                cmd.Parameters.AddWithValue("endDate", dtpEnd.Value.Date);
                NpgsqlDataAdapter da = new NpgsqlDataAdapter(cmd);
                DataTable dt = new DataTable();
                da.Fill(dt);

                dgvReport.DataSource = dt;
                dgvReport.Columns["employee_name"].HeaderText = "Сотрудник";
                dgvReport.Columns["taxable_sum"].HeaderText = "Облагаемая сумма";
                dgvReport.Columns["tax_sum"].HeaderText = "Сумма налога (13%)";
                dgvReport.Columns["taxable_sum"].DefaultCellStyle.Format = "N2";
                dgvReport.Columns["tax_sum"].DefaultCellStyle.Format = "N2";
            }
            catch (Exception ex)
            {
                MessageBox.Show("Ошибка загрузки отчёта: " + ex.Message);
            }
        }

        private void BtnExportExcel_Click(object sender, EventArgs e)

using Npgsql;
using System;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Windows.Forms;
using System.Windows.Forms.DataVisualization.Charting;

namespace MyZadacha
{
    public class PieChartForm : Form
    {
        private NpgsqlConnection con;
        private DateTimePicker dtpStart;
        private DateTimePicker dtpEnd;
        private CheckedListBox clbCostItems;
        private Button btnBuild;
        private Chart chart;
        private DataGridView dgvDetail;

        public PieChartForm(NpgsqlConnection connection)
        {
            con = connection;
            InitializeComponent();
            LoadCostItems();
        }

        private void InitializeComponent()
        {
            this.Text = "Круговая диаграмма расходов по сотрудникам (выбранные статьи)";
            this.Size = new Size(900, 700);
            this.StartPosition = FormStartPosition.CenterParent;

            // Период
            Label lblPeriod = new Label { Text = "Период:", Location = new Point(20, 20), Size = new Size(50, 25) };
            dtpStart = new DateTimePicker { Location = new Point(80, 20), Size = new Size(120, 25), Format = DateTimePickerFormat.Short };
            Label lblTo = new Label { Text = "—", Location = new Point(210, 20), Size = new Size(20, 25) };
            dtpEnd = new DateTimePicker { Location = new Point(240, 20), Size = new Size(120, 25), Format = DateTimePickerFormat.Short };

            // Статьи затрат
            Label lblItems = new Label { Text = "Статьи затрат:", Location = new Point(20, 60), Size = new Size(100, 25) };
            clbCostItems = new CheckedListBox
            {
                Location = new Point(20, 90),
                Size = new Size(250, 200),
                CheckOnClick = true
            };

            // Кнопка
            btnBuild = new Button
            {
                Text = "Построить диаграмму",
                Location = new Point(20, 310),
                Size = new Size(150, 30),
                BackColor = Color.LightGreen
            };
            btnBuild.Click += BtnBuild_Click;

            // Диаграмма
            chart = new Chart
            {
                Location = new Point(300, 20),
                Size = new Size(550, 400),
                BackColor = Color.White
            };
            ChartArea chartArea = new ChartArea("MainArea");
            chart.ChartAreas.Add(chartArea);
            chart.Titles.Add("Расходы по сотрудникам");

            // Детализация (DataGridView)
            dgvDetail = new DataGridView
            {
                Location = new Point(20, 360),
                Size = new Size(830, 280),
                AllowUserToAddRows = false,
                ReadOnly = true,
                AutoSizeColumnsMode = DataGridViewAutoSizeColumnsMode.Fill
            };

            Controls.Add(lblPeriod);
            Controls.Add(dtpStart);
            Controls.Add(lblTo);
            Controls.Add(dtpEnd);
            Controls.Add(lblItems);
            Controls.Add(clbCostItems);
            Controls.Add(btnBuild);
            Controls.Add(chart);
            Controls.Add(dgvDetail);
        }

        private void LoadCostItems()
        {
            try
            {
                string sql = "SELECT id, name FROM costitem ORDER BY name";
                NpgsqlDataAdapter da = new NpgsqlDataAdapter(sql, con);
                DataTable dt = new DataTable();
                da.Fill(dt);
                foreach (DataRow row in dt.Rows)
                {
                    clbCostItems.Items.Add(new { Id = row["id"], Name = row["name"].ToString() });
                }
                // По умолчанию выбрать все
                for (int i = 0; i < clbCostItems.Items.Count; i++)
                    clbCostItems.SetItemChecked(i, true);
            }
            catch (Exception ex)
            {
                MessageBox.Show("Ошибка загрузки статей: " + ex.Message);
            }
        }

        private void BtnBuild_Click(object sender, EventArgs e)
        {
            if (clbCostItems.CheckedItems.Count == 0)
            {
                MessageBox.Show("Выберите хотя бы одну статью затрат.");
                return;
            }
            BuildPieChart();
        }

        private void BuildPieChart()
        {
            DataTable dt = GetData();
            if (dt == null || dt.Rows.Count == 0)
            {
                MessageBox.Show("Нет данных за выбранный период.");
                return;
            }

            // Агрегация по сотрудникам
            var empData = dt.AsEnumerable()
                .GroupBy(r => r["employee_name"].ToString())
                .Select(g => new { Employee = g.Key, Total = g.Sum(r => Convert.ToDecimal(r["total_amount"])) })
                .OrderByDescending(x => x.Total)
                .ToList();

            // Очистка диаграммы
            chart.Series.Clear();
            Series series = new Series("Сумма")
            {
                ChartType = SeriesChartType.Pie,
                IsValueShownAsLabel = true,
                LabelFormat = "#,##0.00",
                LegendText = "#VALX",
                ToolTip = "#VALX: #VALY руб."
            };
            chart.Series.Add(series);

            // Добавление точек
            foreach (var emp in empData)
            {
                series.Points.AddXY(emp.Employee, emp.Total);
            }

            // Отображение процентов
            foreach (DataPoint point in series.Points)
            {
                point.Label = $"{point.AxisLabel}: {point.YValues[0]:N2} руб. ({point.YValues[0] / series.Points.Sum(p => p.YValues[0]) * 100:F1}%)";
                point.LegendText = point.AxisLabel;
            }

            // Детализация в DataGridView
            dgvDetail.DataSource = dt;
            dgvDetail.Columns["cost_item_name"].HeaderText = "Статья затрат";
            dgvDetail.Columns["employee_name"].HeaderText = "Сотрудник";
            dgvDetail.Columns["total_amount"].HeaderText = "Сумма";
            dgvDetail.Columns["total_amount"].DefaultCellStyle.Format = "N2";
        }

        private DataTable GetData()
        {
            DataTable result = new DataTable();
            try
            {
                var selectedIds = clbCostItems.CheckedItems
                    .Cast<dynamic>()
                    .Select(x => x.Id.ToString())
                    .ToArray();
                if (selectedIds.Length == 0) return result;
                string ids = string.Join(",", selectedIds);

                string sql = $@"
                    SELECT 
                        ci.name AS cost_item_name,
                        e.name AS employee_name,
                        SUM(arl.quantity * ci.price) AS total_amount
                    FROM advancereportline arl
                    JOIN costitem ci ON ci.id = arl.cost_item_id
                    JOIN advancereport ar ON ar.id = arl.report_id
                    JOIN employee e ON e.id = ar.employee_id
                    WHERE ar.report_date::date BETWEEN :start AND :end
                      AND ci.id IN ({ids})
                    GROUP BY ci.name, e.name
                    ORDER BY e.name, ci.name";

                NpgsqlCommand cmd = new NpgsqlCommand(sql, con);
                cmd.Parameters.AddWithValue("start", dtpStart.Value.Date);
                cmd.Parameters.AddWithValue("end", dtpEnd.Value.Date);
                NpgsqlDataAdapter da = new NpgsqlDataAdapter(cmd);
                da.Fill(result);
            }
            catch (Exception ex)
            {
                MessageBox.Show("Ошибка данных: " + ex.Message);
                return null;
            }
            return result;
        }
    }
