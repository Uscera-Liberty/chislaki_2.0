ПОНЯЛ! НЕ ПО ДАТЕ, А ПРОСТО ПО НАЛИЧИЮ ЗАПИСИ В cashpayment

Логика:

· Если у сотрудника есть хотя бы одна выплата по этому отчёту (cashpayment) → облагается налогом
· Если нет выплат → налога нет

Даты выплаты НЕ ПРОВЕРЯЕМ вообще.

1. ReportForm.cs – метод Update()

```csharp
public void Update()
{
    // Логика: налог = total_spent * 0.13, если существует выплата (cashpayment) по этому отчёту
    string sql = @"
        SELECT 
            ar.id,
            e.name AS employee_name,
            ar.report_date,
            ar.period_start,
            ar.period_end,
            ar.purpose,
            ar.total_spent,
            COALESCE((
                SELECT SUM(cp.amount)
                FROM cashpayment cp
                WHERE cp.employee_id = ar.employee_id
            ), 0) AS total_paid,
            CASE 
                WHEN EXISTS (
                    SELECT 1 FROM cashpayment cp
                    WHERE cp.employee_id = ar.employee_id
                ) THEN ar.total_spent
                ELSE 0
            END AS taxable_sum,
            CASE 
                WHEN EXISTS (
                    SELECT 1 FROM cashpayment cp
                    WHERE cp.employee_id = ar.employee_id
                ) THEN ROUND(ar.total_spent * 0.13, 2)
                ELSE 0
            END AS tax_sum
        FROM advancereport ar
        JOIN employee e ON e.id = ar.employee_id
        ORDER BY ar.report_date DESC";

    NpgsqlDataAdapter da = new NpgsqlDataAdapter(sql, con);
    ds.Reset();
    da.Fill(ds);
    dt = ds.Tables[0];
    dataGridView1.DataSource = dt;
    
    dataGridView1.Columns[0].HeaderText = "Номер";
    dataGridView1.Columns[1].HeaderText = "Сотрудник";
    dataGridView1.Columns[2].HeaderText = "Дата создания";
    dataGridView1.Columns[3].HeaderText = "Начало периода";
    dataGridView1.Columns[4].HeaderText = "Конец периода";
    dataGridView1.Columns[5].HeaderText = "Цель";
    dataGridView1.Columns[6].HeaderText = "Потрачено";
```

2. ReportExcel.cs – метод GetReportData()

```csharp
private DataTable GetReportData()
{
    DataTable result = new DataTable();
    try
    {
        // Логика: облагаемая сумма = total_spent, если у сотрудника есть выплата (cashpayment)
        // Дата выплаты НЕ ПРОВЕРЯЕТСЯ
        string sql = @"
            WITH employee_payments AS (
                SELECT 
                    e.id AS employee_id,
                    e.name AS employee_name,
                    COALESCE(SUM(cp.amount), 0) AS total_paid,
                    CASE 
                        WHEN COUNT(cp.id) > 0 THEN 1 
                        ELSE 0 
                    END AS has_payment
                FROM employee e
                LEFT JOIN cashpayment cp ON cp.employee_id = e.id
                GROUP BY e.id, e.name
            )
            SELECT 
                ep.employee_name,
                COALESCE(SUM(ar.total_spent), 0) AS total_spent,
                ep.total_paid,
                CASE 
                    WHEN ep.has_payment = 1 THEN COALESCE(SUM(ar.total_spent), 0)
                    ELSE 0
                END AS taxable_sum,
                CASE 
                    WHEN ep.has_payment = 1 THEN ROUND(COALESCE(SUM(ar.total_spent), 0) * 0.13, 2)
                    ELSE 0
                END AS tax_sum
            FROM employee_payments ep
            LEFT JOIN advancereport ar ON ar.employee_id = ep.employee_id
                AND ar.report_date BETWEEN :startDate AND :endDate
            GROUP BY ep.employee_name, ep.total_paid, ep.has_payment
            HAVING COALESCE(SUM(ar.total_spent), 0) > 0
            ORDER BY ep.employee_name";

        NpgsqlCommand cmd = new NpgsqlCommand(sql, con);
        cmd.Parameters.AddWithValue("startDate", dtpStart.Value.Date);
        cmd.Parameters.AddWithValue("endDate", dtpEnd.Value.Date);
        NpgsqlDataAdapter da = new NpgsqlDataAdapter(cmd);
        da.Fill(result);
    }
    catch (Exception ex)
    {
        MessageBox.Show("Ошибка загрузки данных:\n" + ex.Message, "Ошибка", MessageBoxButtons.OK, MessageBoxIcon.Error);
        return null;
    }
    return result;
}
```

Теперь вы можете проверить: добавить вручную запись в cashpayment для сотрудника – сразу появится налог. Удалить запись – налог исчезнет.

Проверка простая:

· Есть строка в cashpayment с employee_id = сотрудник → налог есть
· Нет строки → налога нет
