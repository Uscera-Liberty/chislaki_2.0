ПОНЯЛ! Нужно: облагается налогом РАЗНИЦА между потраченным и выданным

Правильная логика:

· Если выдали 1000, а потратили 1500 → облагается 500 (разница)
· Если выдали 1500, а потратили 1000 → облагается 0 (переплата не облагается)
· Если выдали 0 → облагается вся сумма потраченного

Формула: taxable = MAX(0, total_spent - total_paid)

1. ReportForm.cs – метод Update()

```csharp
public void Update()
{
    // Логика: облагается разница между потраченным и выданным (если потрачено > выдано)
    string sql = @"
        SELECT 
            ar.id,
            e.name AS employee_name,
            ar.report_date,
            ar.period_start,
            ar.period_end,
            ar.purpose,
            ar.total_spent,
            COALESCE((
                SELECT SUM(cp.amount)
                FROM cashpayment cp
                WHERE cp.employee_id = ar.employee_id
            ), 0) AS total_paid,
            CASE 
                WHEN ar.total_spent > COALESCE((
                    SELECT SUM(cp.amount)
                    FROM cashpayment cp
                    WHERE cp.employee_id = ar.employee_id
                ), 0) THEN ar.total_spent - COALESCE((
                    SELECT SUM(cp.amount)
                    FROM cashpayment cp
                    WHERE cp.employee_id = ar.employee_id
                ), 0)
                ELSE 0
            END AS taxable_sum,
            ROUND(
                CASE 
                    WHEN ar.total_spent > COALESCE((
                        SELECT SUM(cp.amount)
                        FROM cashpayment cp
                        WHERE cp.employee_id = ar.employee_id
                    ), 0) THEN (ar.total_spent - COALESCE((
                        SELECT SUM(cp.amount)
                        FROM cashpayment cp
                        WHERE cp.employee_id = ar.employee_id
                    ), 0)) * 0.13
                    ELSE 0
                END, 2
            ) AS tax_sum
        FROM advancereport ar
        JOIN employee e ON e.id = ar.employee_id
        ORDER BY ar.report_date DESC";

    NpgsqlDataAdapter da = new NpgsqlDataAdapter(sql, con);
    ds.Reset();
    da.Fill(ds);
    dt = ds.Tables[0];
    dataGridView1.DataSource = dt;
    
    dataGridView1.Columns[0].HeaderText = "Номер";
    dataGridView1.Columns[1].HeaderText = "Сотрудник";
    dataGridView1.Columns[2].HeaderText = "Дата создания";
    dataGridView1.Columns[3].HeaderText = "Начало периода";
    dataGridView1.Columns[4].HeaderText = "Конец периода";
    dataGridView1.Columns[5].HeaderText = "Цель";
    dataGridView1.Columns[6].HeaderText = "Потрачено";
    dataGridView1.Columns[7].HeaderText = "Выдано";
    dataGridView1.Columns[8].HeaderText = "Облагаемая сумма (разница)";
    dataGridView1.Columns[9].HeaderText = "Налог 13%";
    
    this.StartPosition = FormStartPosition.CenterScreen;
}
```

2. ReportExcel.cs – метод GetReportData()

```csharp
private DataTable GetReportData()
{
    DataTable result = new DataTable();
    try
    {
        // Логика: облагается разница MAX(0, total_spent - total_paid)
        string sql = @"
            WITH employee_totals AS (
                SELECT 
                    e.id AS employee_id,
                    e.name AS employee_name,
                    COALESCE(SUM(ar.total_spent), 0) AS total_spent,
                    COALESCE((
                        SELECT SUM(cp.amount)
                        FROM cashpayment cp
                        WHERE cp.employee_id = e.id
                    ), 0) AS total_paid
                FROM employee e
                LEFT JOIN advancereport ar ON ar.employee_id = e.id
                    AND ar.report_date BETWEEN :startDate AND :endDate
                GROUP BY e.id, e.name
            )
            SELECT 
                employee_name,
                total_spent,
                total_paid,
                CASE 
                    WHEN total_spent > total_paid THEN total_spent - total_paid
                    ELSE 0
                END AS taxable_sum,
                ROUND(
                    CASE 
                        WHEN total_spent > total_paid THEN (total_spent - total_paid) * 0.13
                        ELSE 0
                    END, 2
                ) AS tax_sum
            FROM employee_totals
            WHERE total_spent > 0
            ORDER BY employee_name";

        NpgsqlCommand cmd = new NpgsqlCommand(sql, con);
        cmd.Parameters.AddWithValue("startDate", dtpStart.Value.Date);
        cmd.Parameters.AddWithValue("endDate", dtpEnd.Value.Date);
        NpgsqlDataAdapter da = new NpgsqlDataAdapter(cmd);
        da.Fill(result);
    }
    catch (Exception ex)
    {
        MessageBox.Show("Ошибка загрузки данных:\n" + ex.Message, "Ошибка", MessageBoxButtons.OK, MessageBoxIcon.Error);
        return null;
    }
    return result;
}
```

Что изменилось:

Что было Что стало
Если есть выплата → налог 0 Если выплата МЕНЬШЕ потраченного → налог на РАЗНИЦУ
Если нет выплаты → налог на всю сумму Если выплата БОЛЬШЕ или равна потраченному → налог 0
