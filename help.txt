Разбираю все проблемы по порядку.

**Ошибка внешнего ключа при добавлении строки** — скорее всего `cost_item_id` ссылается на `costitem.id` но при пересоздании таблиц данные слетели, или `is_active` ещё в запросе фильтруется.

**Логика налога по заданию:**
```
Потрачено: 500 руб (отчёт подан)
Получено из кассы: 200 руб (ПОСЛЕ даты отчёта)
Облагается налогом: 200 руб (то что получил после)
Налог: 200 * 0.13 = 26 руб

Потрачено: 500, получено: 700 → облагается 500 (не больше потраченного)
Потрачено: 500, выплаты нет → налога нет
Выплата ДО отчёта → налога нет
```

## ШАГ 1 — SQL пересоздание (запусти в pgAdmin полностью)

```sql
-- Чистим всё
DROP TRIGGER IF EXISTS trg_insert_report_line  ON advancereportline;
DROP TRIGGER IF EXISTS trg_delete_report_line  ON advancereportline;
DROP TRIGGER IF EXISTS trg_update_report_line  ON advancereportline;
DROP FUNCTION IF EXISTS fn_insert_report_line();
DROP FUNCTION IF EXISTS fn_delete_report_line();
DROP FUNCTION IF EXISTS fn_update_report_line();

DROP TABLE IF EXISTS cashpayment       CASCADE;
DROP TABLE IF EXISTS advancereportline CASCADE;
DROP TABLE IF EXISTS advancereport     CASCADE;
DROP TABLE IF EXISTS costitem          CASCADE;
DROP TABLE IF EXISTS employee          CASCADE;

-- Сотрудники
CREATE TABLE employee (
    id    SERIAL PRIMARY KEY,
    name  VARCHAR(150) NOT NULL,
    city  VARCHAR(100),
    phone VARCHAR(20)
);

-- Статьи затрат (БЕЗ is_active)
CREATE TABLE costitem (
    id      SERIAL PRIMARY KEY,
    name    VARCHAR(100)  NOT NULL,
    price   NUMERIC(12,2) NOT NULL DEFAULT 0,
    metrics VARCHAR(20)   NOT NULL DEFAULT 'шт'
);

-- Авансовый отчёт (только total_spent триггером, налог считается в запросах)
CREATE TABLE advancereport (
    id           SERIAL PRIMARY KEY,
    employee_id  INTEGER NOT NULL REFERENCES employee(id) ON DELETE CASCADE,
    report_date  DATE    NOT NULL DEFAULT CURRENT_DATE,
    period_start DATE,
    period_end   DATE,
    purpose      VARCHAR(255),
    total_spent  NUMERIC(12,2) NOT NULL DEFAULT 0
);

-- Строки отчёта
CREATE TABLE advancereportline (
    id           SERIAL PRIMARY KEY,
    report_id    INTEGER       NOT NULL
                     REFERENCES advancereport(id) ON DELETE CASCADE,
    cost_item_id INTEGER       NOT NULL
                     REFERENCES costitem(id) ON DELETE RESTRICT,
    quantity     NUMERIC(10,3) NOT NULL DEFAULT 1
);

-- Выплаты из кассы — привязаны к сотруднику, НЕ к отчёту
CREATE TABLE cashpayment (
    id          SERIAL PRIMARY KEY,
    employee_id INTEGER       NOT NULL
                    REFERENCES employee(id) ON DELETE CASCADE,
    pay_date    DATE          NOT NULL DEFAULT CURRENT_DATE,
    amount      NUMERIC(12,2) NOT NULL,
    note        VARCHAR(255)
);

-- Триггер INSERT строки → прибавить к total_spent
CREATE OR REPLACE FUNCTION fn_insert_report_line()
RETURNS TRIGGER AS $$
DECLARE
    v_price NUMERIC(12,2);
BEGIN
    SELECT price INTO v_price FROM costitem WHERE id = NEW.cost_item_id;
    UPDATE advancereport
    SET total_spent = total_spent + (NEW.quantity * v_price)
    WHERE id = NEW.report_id;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_insert_report_line
AFTER INSERT ON advancereportline
FOR EACH ROW EXECUTE FUNCTION fn_insert_report_line();

-- Триггер DELETE строки → вычесть из total_spent
CREATE OR REPLACE FUNCTION fn_delete_report_line()
RETURNS TRIGGER AS $$
DECLARE
    v_price NUMERIC(12,2);
BEGIN
    SELECT price INTO v_price FROM costitem WHERE id = OLD.cost_item_id;
    UPDATE advancereport
    SET total_spent = GREATEST(0, total_spent - (OLD.quantity * v_price))
    WHERE id = OLD.report_id;
    RETURN OLD;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_delete_report_line
AFTER DELETE ON advancereportline
FOR EACH ROW EXECUTE FUNCTION fn_delete_report_line();

-- Триггер UPDATE строки → пересчитать total_spent
CREATE OR REPLACE FUNCTION fn_update_report_line()
RETURNS TRIGGER AS $$
DECLARE
    v_price NUMERIC(12,2);
BEGIN
    SELECT price INTO v_price FROM costitem WHERE id = NEW.cost_item_id;
    UPDATE advancereport
    SET total_spent = GREATEST(0,
        total_spent - (OLD.quantity * v_price) + (NEW.quantity * v_price))
    WHERE id = NEW.report_id;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_update_report_line
AFTER UPDATE ON advancereportline
FOR EACH ROW EXECUTE FUNCTION fn_update_report_line();
```

## ШАГ 2 — ReportForm.cs — заменить Update()

```csharp
public void Update()
{
    // ПРАВИЛЬНАЯ ЛОГИКА ПО ЗАДАНИЮ:
    // Работник СНАЧАЛА потратил свои деньги (подал отчёт),
    // ПОТОМ получил деньги из кассы (pay_date > report_date).
    // Облагается налогом: сумма выплат которые пришли ПОСЛЕ отчёта,
    // но не больше чем потрачено.
    // Если выплата пришла ДО отчёта — это аванс, налога нет.
    // Если выплат после отчёта нет вообще — налога нет.
    String sql = @"
        SELECT
            ar.id,
            e.name          AS employee_name,
            ar.report_date,
            ar.period_start,
            ar.period_end,
            ar.purpose,
            ar.total_spent,
            -- Выплаты ДО даты отчёта (аванс)
            COALESCE((
                SELECT SUM(cp.amount)
                FROM cashpayment cp
                WHERE cp.employee_id = ar.employee_id
                  AND cp.pay_date <= ar.report_date
            ), 0) AS total_paid,
            -- Облагаемая сумма = выплаты ПОСЛЕ отчёта, но не больше потраченного
            LEAST(
                ar.total_spent,
                COALESCE((
                    SELECT SUM(cp.amount)
                    FROM cashpayment cp
                    WHERE cp.employee_id = ar.employee_id
                      AND cp.pay_date > ar.report_date
                ), 0)
            ) AS taxable_sum,
            -- Налог 13% от облагаемой суммы
            ROUND(
                LEAST(
                    ar.total_spent,
                    COALESCE((
                        SELECT SUM(cp.amount)
                        FROM cashpayment cp
                        WHERE cp.employee_id = ar.employee_id
                          AND cp.pay_date > ar.report_date
                    ), 0)
                ) * 0.13
            , 2) AS tax_sum
        FROM advancereport ar
        JOIN employee e ON e.id = ar.employee_id
        ORDER BY ar.report_date DESC";

    NpgsqlDataAdapter da = new NpgsqlDataAdapter(sql, con);
    ds.Reset();
    da.Fill(ds);
    dt = ds.Tables[0];
    dataGridView1.DataSource = dt;
    dataGridView1.Columns[0].HeaderText = "Номер";
    dataGridView1.Columns[1].HeaderText = "Сотрудник";
    dataGridView1.Columns[2].HeaderText = "Дата создания";
    dataGridView1.Columns[3].HeaderText = "Начало периода";
    dataGridView1.Columns[4].HeaderText = "Конец периода";
    dataGridView1.Columns[5].HeaderText = "Цель";
    dataGridView1.Columns[6].HeaderText = "Потрачено";
    dataGridView1.Columns[7].HeaderText = "Аванс (до отчёта)";
    dataGridView1.Columns[8].HeaderText = "Облагаемая сумма";
    dataGridView1.Columns[9].HeaderText = "Налог (13%)";
    this.StartPosition = FormStartPosition.CenterScreen;
}
```

```csharp
private DataTable GetReportData()
{
    DataTable result = new DataTable();
    try
    {
        // Та же логика что в ReportForm,
        // но группируем по сотруднику за выбранный период
        string sql = @"
            SELECT
                e.name              AS employee_name,
                SUM(ar.total_spent) AS total_spent,
                -- Аванс (выплаты ДО даты каждого отчёта)
                COALESCE((
                    SELECT SUM(cp.amount)
                    FROM cashpayment cp
                    WHERE cp.employee_id = e.id
                      AND cp.pay_date <= MAX(ar.report_date)
                      AND cp.pay_date BETWEEN :startDate AND :endDate
                ), 0) AS total_paid,
                -- Облагаемая: выплаты ПОСЛЕ отчёта, не больше потраченного
                SUM(
                    LEAST(
                        ar.total_spent,
                        COALESCE((
                            SELECT SUM(cp.amount)
                            FROM cashpayment cp
                            WHERE cp.employee_id = ar.employee_id
                              AND cp.pay_date > ar.report_date
                              AND cp.pay_date BETWEEN :startDate AND :endDate
                        ), 0)
                    )
                ) AS taxable_sum,
                -- Налог 13%
                ROUND(SUM(
                    LEAST(
                        ar.total_spent,
                        COALESCE((
                            SELECT SUM(cp.amount)
                            FROM cashpayment cp
                            WHERE cp.employee_id = ar.employee_id
                              AND cp.pay_date > ar.report_date
                              AND cp.pay_date BETWEEN :startDate AND :endDate
                        ), 0)
                    )
                ) * 0.13, 2) AS tax_sum
            FROM advancereport ar
            JOIN employee e ON e.id = ar.employee_id
            WHERE ar.report_date BETWEEN :startDate AND :endDate
            GROUP BY e.id, e.name
            ORDER BY e.name";

        NpgsqlCommand cmd = new NpgsqlCommand(sql, con);
        cmd.Parameters.AddWithValue("startDate", dtpStart.Value.Date);
        cmd.Parameters.AddWithValue("endDate",   dtpEnd.Value.Date);
        NpgsqlDataAdapter da = new NpgsqlDataAdapter(cmd);
        da.Fill(result);
    }
    catch (Exception ex)
    {
        MessageBox.Show("Ошибка загрузки данных:\n" + ex.Message,
            "Ошибка", MessageBoxButtons.OK, MessageBoxIcon.Error);
        return null;
    }
    return result;
}
```
