Перечитываю задание внимательно:

> **"В случае если работник вначале отчитался за потраченные деньги, а потом получил их из кассы — такие суммы облагаются подоходным налогом по ставке 13%"**

---

## Правильная логика

```
Сотрудник СНАЧАЛА потратил свои деньги
    → подал отчёт (есть запись в advancereport)
    → ПОТОМ получил деньги из кассы (cashpayment.pay_date > advancereport.report_date)
    → ЭТО облагается налогом 13%

Сотрудник СНАЧАЛА получил аванс из кассы
    → потом потратил и отчитался
    → НЕ облагается налогом
```

**Ключевое условие:** `cashpayment.pay_date > advancereport.report_date`

Именно это сравнение дат и определяет облагается ли сумма налогом. Сейчас у тебя выплата вообще не привязана к отчёту — значит нужно сравнивать по сотруднику: **если дата выплаты позже даты отчёта сотрудника**.

## ШАГ 1 — ReportForm.cs — заменить метод Update()

```csharp
public void Update()
{
    // ПРАВИЛЬНАЯ ЛОГИКА:
    // Налог = выплаты которые пришли ПОСЛЕ даты отчёта сотрудника
    // То есть: cp.pay_date > ar.report_date
    // Сначала потратил (отчёт) → потом получил (выплата) = налог
    String sql = @"
        SELECT 
            ar.id,
            e.name        AS employee_name,
            ar.report_date,
            ar.period_start,
            ar.period_end,
            ar.purpose,
            ar.total_spent,
            COALESCE((
                SELECT SUM(cp.amount)
                FROM cashpayment cp
                WHERE cp.employee_id = ar.employee_id
            ), 0) AS total_paid,
            COALESCE((
                SELECT SUM(cp.amount)
                FROM cashpayment cp
                WHERE cp.employee_id = ar.employee_id
                  AND cp.pay_date > ar.report_date
            ), 0) AS taxable_sum,
            ROUND(COALESCE((
                SELECT SUM(cp.amount)
                FROM cashpayment cp
                WHERE cp.employee_id = ar.employee_id
                  AND cp.pay_date > ar.report_date
            ), 0) * 0.13, 2) AS tax_sum
        FROM advancereport ar
        JOIN employee e ON e.id = ar.employee_id
        ORDER BY ar.report_date DESC";

    NpgsqlDataAdapter da = new NpgsqlDataAdapter(sql, con);
    ds.Reset();
    da.Fill(ds);
    dt = ds.Tables[0];
    dataGridView1.DataSource = dt;
    dataGridView1.Columns[0].HeaderText = "Номер";
    dataGridView1.Columns[1].HeaderText = "Сотрудник";
    dataGridView1.Columns[2].HeaderText = "Дата создания";
    dataGridView1.Columns[3].HeaderText = "Начало периода";
    dataGridView1.Columns[4].HeaderText = "Конец периода";
    dataGridView1.Columns[5].HeaderText = "Цель";
    dataGridView1.Columns[6].HeaderText = "Потрачено";
    dataGridView1.Columns[7].HeaderText = "Выдано";
    dataGridView1.Columns[8].HeaderText = "Облагаемая сумма";
    dataGridView1.Columns[9].HeaderText = "Налог (13%)";
    this.StartPosition = FormStartPosition.CenterScreen;
}
```

## ШАГ 2 — ReportExcel.cs — заменить метод GetReportData()

```csharp
private DataTable GetReportData()
{
    DataTable result = new DataTable();
    try
    {
        // ПРАВИЛЬНАЯ ЛОГИКА:
        // Облагаемая сумма = выплаты ПОСЛЕ даты отчёта сотрудника
        // cp.pay_date > ar.report_date означает:
        //   сначала потратил свои → подал отчёт → потом получил из кассы
        string sql = @"
            SELECT
                e.name AS employee_name,
                SUM(ar.total_spent) AS total_spent,
                COALESCE((
                    SELECT SUM(cp.amount)
                    FROM cashpayment cp
                    WHERE cp.employee_id = e.id
                      AND cp.pay_date BETWEEN :startDate AND :endDate
                ), 0) AS total_paid,
                COALESCE((
                    SELECT SUM(cp.amount)
                    FROM cashpayment cp
                    WHERE cp.employee_id = e.id
                      AND cp.pay_date > MIN(ar.report_date)
                      AND cp.pay_date BETWEEN :startDate AND :endDate
                ), 0) AS taxable_sum,
                ROUND(COALESCE((
                    SELECT SUM(cp.amount)
                    FROM cashpayment cp
                    WHERE cp.employee_id = e.id
                      AND cp.pay_date > MIN(ar.report_date)
                      AND cp.pay_date BETWEEN :startDate AND :endDate
                ), 0) * 0.13, 2) AS tax_sum
            FROM advancereport ar
            JOIN employee e ON e.id = ar.employee_id
            WHERE ar.report_date BETWEEN :startDate AND :endDate
            GROUP BY e.id, e.name
            ORDER BY e.name";

        NpgsqlCommand cmd = new NpgsqlCommand(sql, con);
        cmd.Parameters.AddWithValue("startDate", dtpStart.Value.Date);
        cmd.Parameters.AddWithValue("endDate",   dtpEnd.Value.Date);
        NpgsqlDataAdapter da = new NpgsqlDataAdapter(cmd);
        da.Fill(result);
    }
    catch (Exception ex)
    {
        MessageBox.Show("Ошибка загрузки данных:\n" + ex.Message,
            "Ошибка", MessageBoxButtons.OK, MessageBoxIcon.Error);
        return null;
    }
    return result;
}
```
