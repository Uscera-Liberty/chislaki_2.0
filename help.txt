public void Update()
{
    // Налог = total_spent * 0.13, если есть выплата ПОСЛЕ даты отчёта
    string sql = @"
        SELECT 
            ar.id,
            e.name AS employee_name,
            ar.report_date,
            ar.period_start,
            ar.period_end,
            ar.purpose,
            ar.total_spent,
            COALESCE((
                SELECT SUM(cp.amount)
                FROM cashpayment cp
                WHERE cp.employee_id = ar.employee_id
                  AND cp.pay_date <= ar.report_date
            ), 0) AS total_paid,
            CASE 
                WHEN EXISTS (
                    SELECT 1 FROM cashpayment cp
                    WHERE cp.employee_id = ar.employee_id
                      AND cp.pay_date > ar.report_date
                ) THEN ar.total_spent
                ELSE 0
            END AS taxable_sum,
            CASE 
                WHEN EXISTS (
                    SELECT 1 FROM cashpayment cp
                    WHERE cp.employee_id = ar.employee_id
                      AND cp.pay_date > ar.report_date
                ) THEN ROUND(ar.total_spent * 0.13, 2)
                ELSE 0
            END AS tax_sum
        FROM advancereport ar
        JOIN employee e ON e.id = ar.employee_id
        ORDER BY ar.report_date DESC";

    NpgsqlDataAdapter da = new NpgsqlDataAdapter(sql, con);
    ds.Reset();
    da.Fill(ds);
    dt = ds.Tables[0];
    dataGridView1.DataSource = dt;
    
    dataGridView1.Columns[0].HeaderText = "Номер";

private DataTable GetReportData()
{
    DataTable result = new DataTable();

        string sql = @"
            SELECT

                COALESCE((
                    SELECT SUM(cp.amount)
                    FROM cashpayment cp
                    WHERE cp.employee_id = e.id
                ), 0) AS total_paid,
                SUM(
                    CASE 
                        WHEN EXISTS (
                            SELECT 1 FROM cashpayment cp2
                            WHERE cp2.employee_id = ar.employee_id
                              AND cp2.pay_date > ar.report_date
                        ) THEN ar.total_spent
                        ELSE 0
                    END
                ) AS taxable_sum,
                ROUND(SUM(
                    CASE 
                        WHEN EXISTS (
                            SELECT 1 FROM cashpayment cp2
                            WHERE cp2.employee_id = ar.employee_id
                              AND cp2.pay_date > ar.report_date
                        ) THEN ar.total_spent * 0.13
                        ELSE 0
                    END
                ), 2) AS tax_sum
            FROM advancereport ar
            JOIN employee e ON e.id = ar.employee_id
            WHERE ar.report_date BETWEEN :startDate AND :endDate
            GROUP BY e.name
            ORDER BY e.name";
